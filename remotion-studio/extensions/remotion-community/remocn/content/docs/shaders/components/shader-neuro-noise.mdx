## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { ShaderNeuroNoise } from "@/components/remocn/shader-neuro-noise";

const ShaderNeuroNoiseScene = () => (
  <ShaderNeuroNoise speed={1} contrast={0.3} brightness={0.05} />
);

export const RemotionRoot = () => (
  <Composition
    id="ShaderNeuroNoise"
    component={ShaderNeuroNoiseScene}
    durationInFrames={150}
    fps={30}
    width={1280}
    height={720}
  />
);
```

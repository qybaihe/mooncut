## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { DynamicGrid } from "@/components/remocn/dynamic-grid";

const DynamicGridScene = () => (
  <DynamicGrid
    cellSize={40}
    lineColor="#27272a"
    background="#0a0a0a"
    speed={0.5}
    direction="diagonal"
  />
);

export const RemotionRoot = () => (
  <Composition
    id="DynamicGrid"
    component={DynamicGridScene}
    durationInFrames={150}
    fps={30}
    width={1280}
    height={720}
  />
);
```

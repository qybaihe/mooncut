## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { EcosystemConstellation } from "@/components/remocn/ecosystem-constellation";

const EcosystemConstellationScene = () => (
  <EcosystemConstellation
    satelliteCount={6}
    centerLabel="V"
    accentColor="#a855f7"
  />
);

export const RemotionRoot = () => (
  <Composition
    id="EcosystemConstellation"
    component={EcosystemConstellationScene}
    durationInFrames={240}
    fps={30}
    width={1280}
    height={720}
  />
);
```

### With Backdrop

Renders its scene transparent — supply the background via `Backdrop`:

```tsx
import { Backdrop } from "@/components/remocn/backdrop";
import { EcosystemConstellation } from "@/components/remocn/ecosystem-constellation";

<Backdrop fill={{ type: "gradient", value: "radial-gradient(ellipse at center, #14101e 0%, #05030a 75%)" }}>
  <EcosystemConstellation satelliteCount={6} centerLabel="V" accentColor="#a855f7" />
</Backdrop>
```

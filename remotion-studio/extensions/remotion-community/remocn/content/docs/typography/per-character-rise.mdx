## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { PerCharacterRise } from "@/components/remocn/per-character-rise";

const PerCharacterRiseScene = () => (
  <PerCharacterRise text="One more thing." distance={32} fontSize={72} />
);

export const RemotionRoot = () => (
  <Composition
    id="PerCharacterRise"
    component={PerCharacterRiseScene}
    durationInFrames={60}
    fps={30}
    width={1280}
    height={720}
  />
);
```

### With Backdrop

The component renders transparent — supply the background via `Backdrop`. It ships single-theme; edit the copied file to re-theme colors.

Pair with `Backdrop` to place the text inside a full-frame fill with a rounded, shadowed content frame:

```tsx
import { Composition } from "remotion";
import { PerCharacterRise } from "@/components/remocn/per-character-rise";
import { Backdrop } from "@/components/remocn/backdrop";

const PerCharacterRiseScene = () => (
  <Backdrop fill={{ type: "color", value: "#ffffff" }}>
    <PerCharacterRise text="One more thing." distance={32} fontSize={72} />
  </Backdrop>
);

export const RemotionRoot = () => (
  <Composition
    id="PerCharacterRise"
    component={PerCharacterRiseScene}
    durationInFrames={60}
    fps={30}
    width={1280}
    height={720}
  />
);
```
